    <script>
        let musicPlaying = false;
        const music = document.getElementById('memorialMusic');
        const musicBtn = document.querySelector('.music-btn');
       
        function toggleMusic() {
            if (musicPlaying) {
                music.pause();
                musicBtn.textContent = '▶️';
            } else {
                music.play();
                musicBtn.textContent = '⏸️';
            }
            musicPlaying = !musicPlaying;
        }

        function showAnswer(num) {
            const allAnswers = document.getElementsByClassName('answer-content');
            for(let i=0; i<allAnswers.length; i++){
                allAnswers[i].style.display = 'none';
            }
            document.getElementById('answer' + num).style.display = 'block';
        }

        window.onload = function() {
            const sections = document.querySelectorAll('.quiz-section, .timeline-section, .album-section');
            sections.forEach((section, index) => {
                setTimeout(() => {
                    section.style.opacity = '1';
                    section.style.transform = 'translateY(0)';
                }, 200 * index);
            });
            toggleMusic();
        }
    </script>
